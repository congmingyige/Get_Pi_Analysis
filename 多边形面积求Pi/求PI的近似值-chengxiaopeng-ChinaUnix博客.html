    求PI的近似值 2011-04-08 17:42:05

    分类： C/C++

        编写一个C程序，用来求出PI的近似值。
        其实其核心思想，就是利用在单位圆中，放置整多边形，当遍数越多，越接近PI,还要使用一些数学公式，我这里就介绍了，就直接上代码了，代码如下：
    1. #include <string.h>
    2. #include <stdio.h>
    3. #include <math.h>

    5. double getPI(int n);

    7. int main(int argc, char *argv[])
    8. {
    9.   int n;
    10.   double PI;
    11.   printf("please input accuracy\n");
    12.   scanf("%d",&n);
    13.   PI = getPI(n);
    14.   printf("the similar value of PI is \n%f\n",PI);

    16.   return 0;
    17. }

    19. double getPI(int n)
    20. {
    21.   int div,i=4;
    22.   double b = sqrt(2)/2.0;
    23.   double c = 0.0;
    24.   for(div=0; div<n; div++){
    25.     b = sqrt(2.0 - 2.0*sqrt(1.0 - b*b)) * 0.5;
    26.     i *= 2;
    27.   }
    28.   c = b * i;
    29.   
    30.   return c;
    31. }
